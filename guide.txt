import threading


def start_timer(request):
    try:
        interval = int(request.GET.get("interval", 60))
        unit = request.GET.get("unit", "seconds")

        # convert to seconds
        if unit == "minutes":
            interval *= 60
        elif unit == "hours":
            interval *= 3600

        timer_state["interval"] = interval
        timer_state["running"] = True

        threading.Thread(target=timer_loop, daemon=True).start()

        return JsonResponse({"status": "timer started"})

    except Exception as e:

def run_all_scripts_task():
